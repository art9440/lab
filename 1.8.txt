#include <stdio.h>
#include <math.h>


int main() {
    int floors, flats, numofflat;
    float nument, numfloor, flatsent;
    scanf("%d %d\n", &floors, &flats);
    scanf("%d", &numofflat);
    flatsent = flats * floors;
    nument = ceil(numofflat / flatsent);
    printf("Number of entrance is %d.\n", (int)nument);
    numfloor = ceil((numofflat - flatsent*(nument - 1)) / flats);
    printf("Number of floor is %d.", (int)numfloor);

    return 0;
}
